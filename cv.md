bool GameOver;

int x, y, fruitX, fruitY, score;

int nTail; 
enum eDirection { STOP = 0, LEFT, RIGHT, UP, DOWN }; 
eDirection dir; 

void Game() { 
GameOver = false; 
dir = STOP; 
x = width / 2-1; 
y = height / 2-1; 
fruitX = rand() % width; 
fruitY = rand() % height; 
score = 0; 
}

cout « "$";

cout « "0"; 
else 
{ 
bool print = false; 
for (int k = 0; k < nTail; k++) 
{

{ 
cout « "$";

int prevY = tailY[0];
